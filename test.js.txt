import express from 'express';

const app = express();
app.use(express.json());

// This will handle the data sent from GHL
app.post("/api/availability", (req, res) => {
  const { date, serviceId, pickup, dropoff } = req.body;

  // 1. Log the data so you can see it in your terminal
  console.log("📩 Webhook Received from GHL:");
  console.log(`Date: ${date}, Service: ${serviceId}, From: ${pickup}, To: ${dropoff}`);

  // 2. Logic: Here is where you'd check your database or Google Sheets
  if (!date || !pickup || !dropoff) {
    return res.status(400).json({ status: "error", message: "Missing required fields" });
  }

  // 3. Respond back to GHL (GHL needs a 200 OK to know it worked)
  res.status(200).json({
    status: "success",
    message: "Availability received",
    receivedData: { date, pickup, dropoff }
  });
});
const PORT = 8080;
app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
